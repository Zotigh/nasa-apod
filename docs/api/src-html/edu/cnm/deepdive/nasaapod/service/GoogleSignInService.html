001package edu.cnm.deepdive.nasaapod.service;
002
003import com.google.android.gms.auth.api.signin.GoogleSignIn;
004import com.google.android.gms.auth.api.signin.GoogleSignInAccount;
005import com.google.android.gms.auth.api.signin.GoogleSignInClient;
006import com.google.android.gms.auth.api.signin.GoogleSignInOptions;
007import edu.cnm.deepdive.nasaapod.ApodApplication;
008
009public class GoogleSignInService {
010
011  private GoogleSignInClient client;
012  private GoogleSignInAccount account;
013
014  private GoogleSignInService() {
015    GoogleSignInOptions options =
016        new GoogleSignInOptions.Builder(GoogleSignInOptions.DEFAULT_SIGN_IN)
017            .requestEmail()
018            .requestId()
019            .build();
020    client = GoogleSignIn.getClient(ApodApplication.getInstance(), options);
021  }
022
023  public static GoogleSignInService getInstance() {
024    return InstanceHolder.INSTANCE;
025  }
026
027  public GoogleSignInClient getClient() {
028    return client;
029  }
030
031  public void setClient(GoogleSignInClient client) {
032    this.client = client;
033  }
034
035  public GoogleSignInAccount getAccount() {
036    return account;
037  }
038
039  public void setAccount(GoogleSignInAccount account) {
040    this.account = account;
041  }
042
043  private static class InstanceHolder {
044
045    private static final GoogleSignInService INSTANCE = new GoogleSignInService();
046
047  }
048
049}
