001package edu.cnm.deepdive.nasaapod.model.pojo;
002
003import android.arch.persistence.room.Embedded;
004import android.arch.persistence.room.Relation;
005import edu.cnm.deepdive.nasaapod.model.entity.Access;
006import edu.cnm.deepdive.nasaapod.model.entity.Apod;
007import java.util.List;
008
009public class ApodWithAccesses {
010
011  @Embedded
012  private Apod apod;
013
014  @Relation(parentColumn = "apod_id", entityColumn = "apod_id")
015  private List<Access> accesses;
016
017  public Apod getApod() {
018    return apod;
019  }
020
021  public void setApod(Apod apod) {
022    this.apod = apod;
023  }
024
025  public List<Access> getAccesses() {
026    return accesses;
027  }
028
029  public void setAccesses(List<Access> accesses) {
030    this.accesses = accesses;
031  }
032
033}
